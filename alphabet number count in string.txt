#include<stdio.h>

void main()

{

char str[10];

gets(str);

	int i,count1=0,count2=0,count3=0;


	for(i=0;str[i]!='\0';i++)

	{
		
if(isupper(str[i]))
	
	{
	
		count1=count1+1;
	
	}
	
	if(islower(str[i]))

		{
	
		count2=count2+1;

		}
		
 if(isdigit(str[i]))
	
	{
		
	count3=count3+1;

		}

	}
	printf("Count of uppercase characters: %d\n",count1);

           return 0;
}
